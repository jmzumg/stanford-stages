# Pinned dependencies for the stanford-stages Docker image.
#
# This is a cleaned-up copy of the root requirements.txt. Two entries were
# removed because they are not installable package specs:
#   - pyEDFlib.egg==info   (not a real PyPI package)
#   - skimage==0.0         (scikit_image below already provides the skimage module)
#
# tf_nightly_2.0_preview is intentionally omitted; tensorflow-cpu==2.5.0 already
# supplies everything the code needs.
#
# tensorflow-cpu (not tensorflow): there is no GPU on the compute host, and the
# CPU wheel is smaller. gpflow 2.1.4 hard-requires the 'tensorflow' distribution,
# so it is installed --no-deps in the Dockerfile; its non-tensorflow runtime deps
# (multipledispatch, tabulate, typing-extensions, packaging) are listed here so
# they are still present.

numpy==1.19.5
scipy==1.3.2
scikit_image==0.15.0
PyWavelets==1.0.3
pandas==0.25.1
matplotlib==3.1.1
# NOTE: bumped from h5py==2.10.0 to satisfy tensorflow 2.5.0, which
# requires h5py~=3.1.0. The 2.10.0 pin in the repo's requirements.txt is
# unsatisfiable under pip and only worked via conda install ordering. The
# code uses only basic h5py APIs (File, dataset read/write) that are
# unchanged between 2.x and 3.x, and the saved HDF5 files are readable by
# both versions.
h5py==3.1.0
pyedflib==0.1.22
tensorflow-cpu==2.5.0
tensorflow-probability==0.13.0

# gpflow 2.1.4 non-tensorflow runtime deps (gpflow itself is installed --no-deps).
multipledispatch>=0.6
tabulate
typing-extensions
packaging
